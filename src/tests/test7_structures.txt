#PYTHON
ALGORITHME GestionStructures

STRUCTURE Point
    x : reel;
    y : reel;
FINSTRUCTURE

STRUCTURE Rectangle
    coinSupGauche : Point;
    largeur : reel;
    hauteur : reel;
FINSTRUCTURE

STRUCTURE Etudiant
    nom : chaine;
    age : entier;
    notes : entier[3];
    actif : booleen;
FINSTRUCTURE

FONCTION calculerPerimetre(rect : Rectangle) : reel
DEBUT
    RETOUR 2 * (rect.largeur + rect.hauteur)
FINFONCTION

FONCTION calculerAire(rect : Rectangle) : reel
DEBUT
    RETOUR rect.largeur * rect.hauteur
FINFONCTION

VAR
    p1, p2 : Point;
    rect : Rectangle;
    etud : Etudiant;
    perimetre, aire : reel;
    i : entier;

DEBUT
    // Points avec coordonnées positives
    p1.x <- 0.0
    p1.y <- 0.0

    p2.x <- 5.0
    p2.y <- 3.0

    ECRIRE "Point 1 : (", p1.x, ", ", p1.y, ")"
    ECRIRE "Point 2 : (", p2.x, ", ", p2.y, ")"

    // Rectangle
    rect.coinSupGauche <- p1
    rect.largeur <- 10.5
    rect.hauteur <- 5.2

    ECRIRE "Rectangle : largeur=", rect.largeur, ", hauteur=", rect.hauteur

    perimetre <- calculerPerimetre(rect)
    aire <- calculerAire(rect)
    ECRIRE "Périmètre = ", perimetre
    ECRIRE "Aire = ", aire

    // Étudiant
    etud.nom <- "Alice"
    etud.age <- 20
    etud.actif <- VRAI

    etud.notes[0] <- 15
    etud.notes[1] <- 12
    etud.notes[2] <- 18

    ECRIRE "Étudiant : ", etud.nom
    ECRIRE "Âge : ", etud.age
    ECRIRE "Actif : ", etud.actif

    ECRIRE "Notes : "
    POUR i <- 0 JUSQUA 2 FAIRE
        ECRIRE "  Note ", i, " = ", etud.notes[i]
    FINPOUR

    // Point avec coordonnées négatives
    p1.x <- -5.5
    p1.y <- -3.2
    ECRIRE "Point négatif : (", p1.x, ", ", p1.y, ")"
FIN
