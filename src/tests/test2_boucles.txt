#PYTHON
ALGORITHME TestBoucles
VAR
    i, j, somme, produit : entier;
DEBUT
    // Boucle POUR
    somme <- 0
    POUR i <- 1 JUSQUA 5 FAIRE
        somme <- somme + i
    FINPOUR
    ECRIRE "Somme 1-5 = ", somme
    
    // Boucle TANTQUE
    i <- 1
    produit <- 1
    TANTQUE i <= 5 FAIRE
        produit <- produit * i
        i <- i + 1
    FINTANTQUE
    ECRIRE "Factorielle 5 = ", produit
    
    // Boucle REPETER
    j <- 0
    REPETER
        ECRIRE "Répétition ", j
        j <- j + 1
    JUSQUA j = 3
FIN
