#PYTHON
ALGORITHME GestionProcedures
VAR
    nombre : entier;
    message : chaine;

PROCEDURE afficherMessage(msg : chaine)
DEBUT
    ECRIRE "Message : ", msg
FINPROCEDURE

PROCEDURE afficherTableMulti(n : entier)
VAR
    i : entier;
DEBUT
    ECRIRE "Table de multiplication de ", n
    POUR i <- 1 JUSQUA 10 FAIRE
        ECRIRE n, " x ", i, " = ", n * i
    FINPOUR
FINPROCEDURE

PROCEDURE analyserNombre(n : entier)
DEBUT
    SI n > 0 ALORS
        ECRIRE n, " est positif"
    SINON
        SI n < 0 ALORS
            ECRIRE n, " est négatif"
        SINON
            ECRIRE "Le nombre est zéro"
        FINSI
    FINSI
FINPROCEDURE

DEBUT
    message <- "Bienvenue"
    afficherMessage(message)

    nombre <- 7
    afficherTableMulti(nombre)

    nombre <- -5
    afficherTableMulti(nombre)

    analyserNombre(15)
    analyserNombre(-8)
    analyserNombre(0)
FIN
