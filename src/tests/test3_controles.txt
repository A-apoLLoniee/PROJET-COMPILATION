#PYTHON
ALGORITHME StructuresDecision
VAR
    x, y, nombre : entier;
    estVrai, condition : booleen;
DEBUT
    x <- 15
    y <- 25

    // Condition simple
    SI x < y ALORS
        ECRIRE "x est plus petit"
    FINSI

    // Condition avec SINON
    SI x > y ALORS
        ECRIRE "x est plus grand"
    SINON
        ECRIRE "x n'est pas plus grand"
    FINSI

    // Condition complexe
    estVrai <- VRAI
    condition <- (x > 10) ET estVrai
    SI condition ALORS
        ECRIRE "Condition complexe vraie"
    FINSI

    // Conditions avec nombres négatifs
    nombre <- -15
    SI nombre < 0 ALORS
        ECRIRE nombre, " est négatif"
    SINON
        ECRIRE nombre, " est positif ou nul"
    FINSI

    x <- -10
    y <- -20
    SI x > y ALORS
        ECRIRE x, " est plus grand que ", y
    FINSI
FIN
