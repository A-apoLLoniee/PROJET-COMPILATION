Fonction calculerCarre(x: entier): entier
DEBUT
    RETOUR x * x
FinFonction

VAR
    age, limite, resultat, compteur, i: entier;
    message: chaine;
    estMajeur: booleen;
    prix: reel;
    tab: entier[10];
DEBUT
    limite <- 18;
    age <- 20;
    prix <- 12.5;
    estMajeur <- age >= limite;
    SI estMajeur ET age < 100 ALORS
        message <- "Personne majeure";
        ECRIRE message;
    SINON
        ECRIRE "Personne mineure";
    FINSI
    
    resultat <- age + 5;
    ECRIRE "Age: ", age;
    ECRIRE "Prix: ", prix;
    
    // Test d'appel de fonction
    resultat <- calculerCarre(age);
    ECRIRE "Carré de l'âge: ", resultat;
    
    // Test de boucle POUR
    POUR i <- 1 JUSQUA 5 FAIRE
        ECRIRE "Itération: ", i;
    FINPOUR
    
    // Test de boucle TANTQUE
    compteur <- 0;
    TANTQUE compteur < 3 FAIRE
        ECRIRE "Tantque: ", compteur;
        compteur <- compteur + 1;
